import redis, json, time, uuid

class SignalBus:
    def __init__(self, host="localhost", port=6379):
        self.r = redis.Redis(host=host, port=port, decode_responses=True)
        self.stream = "v4_signals"

    def publish(self, signal: dict):
        signal["signal_id"] = str(uuid.uuid4())
        signal["ts"] = int(time.time())

        self.r.xadd(self.stream, {"data": json.dumps(signal)})
        return signal["signal_id"]

    def consume(self, group="decision_group", consumer="c1"):
        try:
            self.r.xgroup_create(self.stream, group, id="0", mkstream=True)
        except:
            pass

        msgs = self.r.xreadgroup(group, consumer, {self.stream: ">"}, count=10, block=2000)
        results = []

        for stream, items in msgs:
            for msg_id, fields in items:
                data = json.loads(fields["data"])
                results.append((msg_id, data))

        return results

    def ack(self, msg_id, group="decision_group"):
        self.r.xack(self.stream, group, msg_id)
```

class PortfolioManager:

    def __init__(self):
        self.max_positions = 3
        self.current_positions = []

    def allocate(self, signals):
        if len(self.current_positions) >= self.max_positions:
            return None

        best = signals[0]

        size = self._position_size(best)

        order = {
            "symbol": best["symbol"],
            "direction": best["direction"],
            "size": size,
            "sl": best["sl"],
            "tp": best["tp"]
        }

        self.current_positions.append(order)
        return order

    def _position_size(self, signal):
        base = 1.0

        *# 简单风险调整*
        if signal["score"] > 18:
            return base * 1.5
        if signal["score"] < 15:
            return base * 0.5

        return base
```

import requests, time, hmac, hashlib

class BinanceAdapter:
    def __init__(self, api_key="", secret=""):
        self.key = api_key
        self.secret = secret
        self.base = "https://fapi.binance.com"

    def _sign(self, query):
        return hmac.new(self.secret.encode(), query.encode(), hashlib.sha256).hexdigest()

    def place_order(self, symbol, side, qty):
        ts = int(time.time() * 1000)
        query = f"symbol={symbol}&side={side}&type=MARKET&quantity={qty}&timestamp={ts}"
        sig = self._sign(query)

        url = f"{self.base}/fapi/v1/order?{query}&signature={sig}"
        headers = {"X-MBX-APIKEY": self.key}

        r = requests.post(url, headers=headers)
        return r.json()
```

import time

class SmartExecutor:

    def __init__(self, adapter):
        self.adapter = adapter

    def execute(self, order):
        symbol = order["symbol"]
        side = "BUY" if order["direction"] == "LONG" else "SELL"
        size = order["size"]

        *# 分批执行（简化 TWAP）*
        parts = 3
        part_size = size / parts

        results = []

        for _ in range(parts):
            res = self.adapter.place_order(symbol, side, part_size)
            results.append(res)
            time.sleep(1)

        return results
```

class StateStore:

    def __init__(self):
        self.positions = []
        self.balance = 10000
        self.max_drawdown = 0.2
        self.equity_peak = self.balance

    def update_position(self, pos):
        self.positions.append(pos)

    def get_positions(self):
        return self.positions

    def update_equity(self, pnl):
        self.balance += pnl
        self.equity_peak = max(self.equity_peak, self.balance)

    def drawdown(self):
        return (self.equity_peak - self.balance) / self.equity_peak

class RiskManager:

    def __init__(self, state):
        self.exposure = ExposureControl(state)
        self.dd = DrawdownGuard(state)
        self.kill = KillSwitch()

    def allow(self, order):
        if self.kill.triggered:
            return False

        if not self.exposure.check(order):
            return False

        if not self.dd.check():
            self.kill.activate()
            return False

        return True

class SignalEvaluator:

    def evaluate(self, signal, trade_result):
        """
        用结果反推信号质量

        pnl = trade_result["pnl"]

        base_score = signal["score"]

class FactorOptimizer:

    def __init__(self):
        self.weights = {
            "trend": 0.25,
            "volatility": 0.20,
            "liquidity": 0.20,
            "rs": 0.20,
            "micro": 0.15
        }

    def update(self, signal_result):
        pnl = signal_result["pnl"]

        *# 简化梯度更新（核心思想）*
        lr = 0.01

        if pnl > 0:
            self.weights["trend"] += lr
            self.weights["rs"] += lr
        else:
            self.weights["volatility"] -= lr
            self.weights["micro"] -= lr

        self._normalize()

    def _normalize(self):
        s = sum(self.weights.values())
        for k in self.weights:
            self.weights[k] /= s
```

class StrategyPruner:

    def __init__(self):
        self.strategy_stats = {}

    def update(self, signal_eval):
        sid = signal_eval["signal_id"]

        pnl = signal_eval["pnl"]

        if sid not in self.strategy_stats:
            self.strategy_stats[sid] = {"win": 0, "loss": 0}

        if pnl > 0:
            self.strategy_stats[sid]["win"] += 1
        else:
            self.strategy_stats[sid]["loss"] += 1

    def is_bad(self, sid):
        s = self.strategy_stats.get(sid, {})
        total = s.get("win", 0) + s.get("loss", 0)
        if total < 10:
            return False

        winrate = s["win"] / total
        return winrate < 0.4

import numpy as np

class RegimeClassifier:

    def classify(self, klines):
        closes = np.array([float(k[4]) for k in klines])

        if len(closes) < 20:
            return "volatile"

        returns = np.diff(closes) / closes[:-1]
        volatility = np.std(returns)

        trend_strength = abs((closes[-1] - closes[0]) / closes[0])

        *# ===== 核心逻辑 =====*
        if volatility > 0.03:
            return "volatile"

        if trend_strength > 0.05:
            return "trend"

        return "range"
```

## **competition\_engine\.py**

```Python
class StrategyCompetitionEngine:

    def __init__(self, strategies):
        self.strategies = strategies

    def run(self, symbol, data, regime):
        candidates = []

        for s in self.strategies:
            sig = s.generate(symbol, data)
            if sig:
                sig["regime"] = regime
                candidates.append(sig)

        if not candidates:
            return None

        *# ===== 核心：竞争排序 =====*
        best = max(candidates, key=lambda x: x["score"])

        return {
            "winner": best,
            "candidates": candidates
        }
```

    def fuse(self, competition_result):

        total = sum(c["score"] for c in candidates)

    def generate(self, symbol):

        price = closes[-1]

        sl = price * (1 - STOP_LOSS_FIXED)

            "direction": direction,
            "entry": price,

V4.7 Risk Kernel（完整风控内核）
替换 SellManager
"""

import time
import logging

logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(message)s")
logger = logging.getLogger("risk_kernel")


class RiskKernel:

    def __init__(self):
        # 仓位状态
        self.positions = {}

        # 冷却控制
        self.last_action_time = {}

        # 参数（后续可接 config）
        self.stop_loss_pct = 0.02
        self.trailing_activate = 0.015
        self.trailing_drawdown = 0.03
        self.max_holding_seconds = 86400 * 5
        self.cooldown = 60

        logger.info("RiskKernel 初始化完成")

    # =========================
    # 仓位管理
    # =========================

    def open_position(self, symbol, direction, price, size=1):
        self.positions[symbol] = {
            "direction": direction,
            "entry_price": price,
            "size": size,
            "entry_time": time.time(),

            # 浮盈状态
            "peak_price": price,
            "max_profit": 0.0,

            # trailing状态
            "trailing_active": False
        }

        logger.info(f"开仓: {symbol} {direction} @ {price}")

    def close_position(self, symbol, reason):
        if symbol in self.positions:
            logger.info(f"平仓: {symbol} reason={reason}")
            del self.positions[symbol]
            self.last_action_time[symbol] = time.time()

    def has_position(self, symbol):
        return symbol in self.positions

    # =========================
    # 主检查入口
    # =========================

    def evaluate(self, symbol, current_price):
        """
        返回：
        None（不操作）
        或 {action: SELL, reason: xxx}
        """

        if symbol not in self.positions:
            return None

        if not self._cooldown_ok(symbol):
            return None

        pos = self.positions[symbol]
        direction = pos["direction"]
        entry = pos["entry_price"]

        # === 更新浮盈状态 ===
        profit = self._calc_profit(direction, entry, current_price)
        self._update_peak(pos, current_price, profit)

        # === 风控优先级 ===

        # 1. 固定止损
        if self._stop_loss(pos, current_price):
            return self._exit(symbol, "STOP_LOSS")

        # 2. Trailing止盈
        if self._trailing_stop(pos, current_price):
            return self._exit(symbol, "TRAILING_STOP")

        # 3. 持仓时间
        if self._time_stop(pos):
            return self._exit(symbol, "TIME_EXIT")

        return None

    # =========================
    # 风控模块
    # =========================

    def _stop_loss(self, pos, price):
        entry = pos["entry_price"]
        direction = pos["direction"]

        if direction == "LONG":
            return price <= entry * (1 - self.stop_loss_pct)
        else:
            return price >= entry * (1 + self.stop_loss_pct)

    def _trailing_stop(self, pos, price):
        entry = pos["entry_price"]
        peak = pos["peak_price"]
        direction = pos["direction"]

        profit = self._calc_profit(direction, entry, price)

        # 激活 trailing
        if not pos["trailing_active"] and profit >= self.trailing_activate:
            pos["trailing_active"] = True
            logger.info("Trailing 激活")

        if not pos["trailing_active"]:
            return False

        # 回撤判断
        if direction == "LONG":
            drawdown = (peak - price) / peak
        else:
            drawdown = (price - peak) / peak

        return drawdown >= self.trailing_drawdown

    def _time_stop(self, pos):
        return time.time() - pos["entry_time"] >= self.max_holding_seconds

    # =========================
    # 状态更新
    # =========================

    def _update_peak(self, pos, price, profit):
        direction = pos["direction"]

        if direction == "LONG":
            if price > pos["peak_price"]:
                pos["peak_price"] = price
        else:
            if price < pos["peak_price"]:
                pos["peak_price"] = price

        pos["max_profit"] = max(pos["max_profit"], profit)

    def _calc_profit(self, direction, entry, price):
        if direction == "LONG":
            return (price - entry) / entry
        else:
            return (entry - price) / entry

    # =========================
    # 执行控制
    # =========================

    def _exit(self, symbol, reason):
        self.last_action_time[symbol] = time.time()
        return {
            "symbol": symbol,
            "action": "SELL",
            "reason": reason
        }

    def _cooldown_ok(self, symbol):
        if symbol not in self.last_action_time:
            return True
        return time.time() - self.last_action_time[symbol] > self.cooldown

"""
V4.7 Risk Kernel v2
- 资金管理（risk budget + sizing）
- 爆仓控制（简化 liquidation model）
- 组合风险控制（portfolio VaR-like）
"""

import time
import logging
import math

logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(message)s")
logger = logging.getLogger("risk_kernel_v2")


class RiskKernel:

    def __init__(self, capital=100000):
        # 资金
        self.initial_capital = capital
        self.capital = capital

        # 仓位
        self.positions = {}

        # 风控参数
        self.risk_per_trade = 0.01          # 单笔风险 1%
        self.max_portfolio_risk = 0.05      # 总风险 5%
        self.max_leverage = 5

        self.stop_loss_pct = 0.02
        self.trailing_activate = 0.015
        self.trailing_drawdown = 0.03

        self.cooldown = 30
        self.last_action_time = {}

        # Kill Switch
        self.max_drawdown = 0.25
        self.equity_peak = capital
        self.kill_switch = False

        logger.info("RiskKernel v2 初始化完成")

    # =========================
    # 开仓（带仓位 sizing）
    # =========================

    def open_position(self, symbol, direction, price, volatility):
        if self.kill_switch:
            logger.warning("Kill Switch 激活，禁止开仓")
            return None

        size = self._calc_position_size(price, volatility)

        if size <= 0:
            return None

        liquidation_price = self._estimate_liquidation(price, direction)

        self.positions[symbol] = {
            "direction": direction,
            "entry_price": price,
            "size": size,
            "entry_time": time.time(),

            "peak_price": price,
            "trailing_active": False,

            "liquidation": liquidation_price
        }

        logger.info(f"开仓 {symbol} {direction} size={round(size,4)} liq={round(liquidation_price,2)}")
        return size

    # =========================
    # 主评估
    # =========================

    def evaluate(self, symbol, price):
        if symbol not in self.positions:
            return None

        if not self._cooldown_ok(symbol):
            return None

        pos = self.positions[symbol]
        direction = pos["direction"]

        # 更新 peak
        self._update_peak(pos, price)

        # 1. 爆仓保护（提前退出）
        if self._near_liquidation(pos, price):
            return self._exit(symbol, "LIQUIDATION_RISK")

        # 2. 固定止损
        if self._stop_loss(pos, price):
            return self._exit(symbol, "STOP_LOSS")

        # 3. Trailing
        if self._trailing(pos, price):
            return self._exit(symbol, "TRAILING")

        return None

    # =========================
    # 平仓后更新资金
    # =========================

    def close_position(self, symbol, price):
        if symbol not in self.positions:
            return

        pos = self.positions[symbol]
        pnl = self._calc_pnl(pos, price)

        self.capital += pnl

        # 更新 equity peak
        self.equity_peak = max(self.equity_peak, self.capital)

        # drawdown 检查
        dd = (self.equity_peak - self.capital) / self.equity_peak
        if dd > self.max_drawdown:
            self.kill_switch = True
            logger.error("触发 Kill Switch（最大回撤）")

        logger.info(f"平仓 {symbol} pnl={round(pnl,2)} capital={round(self.capital,2)}")

        del self.positions[symbol]
        self.last_action_time[symbol] = time.time()

    # =========================
    # 资金管理
    # =========================

    def _calc_position_size(self, price, vol):
        """
        基于波动率的仓位 sizing
        """
        risk_amount = self.capital * self.risk_per_trade

        if vol <= 0:
            return 0

        stop_distance = price * self.stop_loss_pct

        size = risk_amount / stop_distance

        # 限制最大杠杆
        notional = size * price
        if notional > self.capital * self.max_leverage:
            size = (self.capital * self.max_leverage) / price

        # 组合风险限制
        if self._portfolio_risk() > self.max_portfolio_risk:
            return 0

        return size

    def _portfolio_risk(self):
        """
        简化组合风险（所有仓位风险叠加）
        """
        total = 0
        for p in self.positions.values():
            risk = p["size"] * p["entry_price"] * self.stop_loss_pct
            total += risk
        return total / self.capital if self.capital > 0 else 0

    # =========================
    # 爆仓模型（简化版）
    # =========================

    def _estimate_liquidation(self, price, direction):
        """
        简化：按最大杠杆估算
        """
        if direction == "LONG":
            return price * (1 - 1 / self.max_leverage)
        else:
            return price * (1 + 1 / self.max_leverage)

    def _near_liquidation(self, pos, price):
        liq = pos["liquidation"]
        if pos["direction"] == "LONG":
            return price <= liq * 1.05
        else:
            return price >= liq * 0.95

    # =========================
    # 止损 / trailing
    # =========================

    def _stop_loss(self, pos, price):
        entry = pos["entry_price"]
        if pos["direction"] == "LONG":
            return price <= entry * (1 - self.stop_loss_pct)
        else:
            return price >= entry * (1 + self.stop_loss_pct)

    def _trailing(self, pos, price):
        entry = pos["entry_price"]
        peak = pos["peak_price"]

        profit = self._calc_profit(pos, price)

        if not pos["trailing_active"] and profit > self.trailing_activate:
            pos["trailing_active"] = True

        if not pos["trailing_active"]:
            return False

        if pos["direction"] == "LONG":
            dd = (peak - price) / peak
        else:
            dd = (price - peak) / peak

        return dd > self.trailing_drawdown

    # =========================
    # 工具函数
    # =========================

    def _update_peak(self, pos, price):
        if pos["direction"] == "LONG":
            pos["peak_price"] = max(pos["peak_price"], price)
        else:
            pos["peak_price"] = min(pos["peak_price"], price)

    def _calc_profit(self, pos, price):
        e = pos["entry_price"]
        if pos["direction"] == "LONG":
            return (price - e) / e
        else:
            return (e - price) / e

    def _calc_pnl(self, pos, price):
        profit = self._calc_profit(pos, price)
        return profit * pos["size"] * pos["entry_price"]

    def _exit(self, symbol, reason):
        return {"symbol": symbol, "action": "SELL", "reason": reason}

    def _cooldown_ok(self, symbol):
        if symbol not in self.last_action_time:
            return True
        return time.time() - self.last_action_time[symbol] > self.cooldown

        # 风险参数
        self.max_risk_ratio = 0.05  # 最大账户风险5%
        self.max_symbol_risk = 0.02
        self.max_leverage = 5

        # 持仓
        self.positions = {}  # symbol -> position dict

        # 价格历史（用于相关性）
        self.price_history = {}

        # 极端行情
        self.vol_threshold = 0.05

    # =============================
    # 1️⃣ 资金管理
    # =============================
    def risk_budget(self):
        return self.balance * self.max_risk_ratio

    def current_risk(self):
        total = 0
        for pos in self.positions.values():
            total += pos["risk"]
        return total

    # =============================
    # 2️⃣ 相关性矩阵
    # =============================
    def update_price(self, symbol, price):
        self.price_history.setdefault(symbol, []).append(price)
        if len(self.price_history[symbol]) > 50:
            self.price_history[symbol].pop(0)

    def correlation(self, sym1, sym2):
        p1 = self.price_history.get(sym1)
        p2 = self.price_history.get(sym2)
        if not p1 or not p2 or len(p1) < 10 or len(p2) < 10:
            return 0.5  # 默认中性

        return np.corrcoef(p1, p2)[0, 1]

    def correlation_penalty(self, symbol):
        penalty = 1.0
        for sym in self.positions:
            if sym == symbol:
                continue
            corr = abs(self.correlation(symbol, sym))
            penalty += corr * 0.5
        return penalty

    # =============================
    # 3️⃣ 动态杠杆
    # =============================
    def dynamic_leverage(self, symbol, volatility, confidence):
        lev = 1.0

        # 波动率降低杠杆
        if volatility > 0.04:
            lev *= 0.5

        # 相关性降低杠杆
        lev /= self.correlation_penalty(symbol)

        # 信号增强
        lev *= (1 + confidence)

        return min(self.max_leverage, max(1, lev))

    # =============================
    # 4️⃣ 爆仓控制
    # =============================
    def liquidation_check(self, entry, current, leverage, direction):
        if direction == "LONG":
            liq_price = entry * (1 - 1 / leverage)
            return current <= liq_price
        else:
            liq_price = entry * (1 + 1 / leverage)
            return current >= liq_price

    # =============================
    # 5️⃣ 极端行情保护
    # =============================
    def extreme_market(self, prices):
        if len(prices) < 5:
            return False
        returns = np.diff(prices) / prices[:-1]
        vol = np.std(returns)
        return vol > self.vol_threshold

    # =============================
    # 6️⃣ 开仓评估
    # =============================
    def evaluate_entry(self, signal, price_series):
        symbol = signal["symbol"]
        confidence = signal.get("confidence", 0.5)

        # 极端行情禁止开仓
        if self.extreme_market(price_series):
            logger.warning(f"{symbol} 极端行情，禁止开仓")
            return None

        volatility = np.std(price_series) / np.mean(price_series)

        leverage = self.dynamic_leverage(symbol, volatility, confidence)

        risk_per_trade = self.balance * self.max_symbol_risk

        # 相关性惩罚
        penalty = self.correlation_penalty(symbol)
        risk_adjusted = risk_per_trade / penalty

        # 总风险限制
        if self.current_risk() + risk_adjusted > self.risk_budget():
            logger.warning("超过总风险预算")
            return None

        position_size = risk_adjusted * leverage

        return {
            "symbol": symbol,
            "size": position_size,
            "leverage": leverage,
            "risk": risk_adjusted
        }

    # =============================
    # 7️⃣ 持仓更新
    # =============================
    def update_position(self, symbol, pos):
        self.positions[symbol] = pos

    # =============================
    # 8️⃣ 全局风控扫描
    # =============================
    def risk_scan(self, market_prices):
        actions = []

        for sym, pos in self.positions.items():
            price = market_prices.get(sym)
            if not price:
                continue

            # 爆仓检查
            if self.liquidation_check(pos["entry"], price, pos["leverage"], pos["direction"]):
                actions.append({
                    "symbol": sym,
                    "action": "FORCE_CLOSE",
                    "reason": "liquidation risk"
                })

        return actions
